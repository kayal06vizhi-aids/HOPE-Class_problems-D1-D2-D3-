CharacterDigitSpecialclass Solution {    public static void main(String[] args) {        String str = "Hello##123";        int alphabet = 0;        int digit = 0;        int special = 0;        for (int i = 0; i < str.length(); i++) {            char ch = str.charAt(i);            if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')) {                alphabet++;            } else if (ch >= '0' && ch <= '9') {                digit++;            } else {                special++;            }        }        System.out.println(alphabet+" "+digit+" "+special);    }}Palindromeclass Solution {    public boolean isPalindrome(String s) {        int l = 0;        int r = s.length() - 1;        while (l < r) {            if (s.charAt(l) != s.charAt(r)) {                return false;            }            l++;            r--;        }        return true;    }}Reverseclass Solution {    public void reverseString(char[] s) {        int start = 0;        int end = s.length - 1;        while(start <= end){            char temp = s[start];            s[start] = s[end];            s[end] = temp;            start++;            end--;        }    }}Toggle Characterclass Solution {    public static void main(String[] args) {        String str = "HelloWorld";        char[] arr = str.toCharArray();        for (int i = 0; i < str.length(); i++) {            char ch = str.charAt(i);            if (ch >= 'a' && ch <= 'z') {                if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {                    vowels++;                } else {                    consonants++;                }            }        }        System.out.println("Vowels: " + vowels);        System.out.println("Consonants: " + consonants);    }}LowerCaseclass Solution {    public static void main(String[] args) {        String str = "helloWorld";        char[] arr = str.toCharArray();        for (int i = 0;
